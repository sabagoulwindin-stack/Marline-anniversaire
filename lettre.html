Chargement du souvenir le plus précieux...


function startMusic() { const audio = document.getElementById('music'); audio.play().then(() => { // Une fois que ça joue, on retire l'écouteur pour ne pas relancer à chaque clic document.removeEventListener('click', startMusic); document.removeEventListener('touchstart', startMusic); }).catch(error => { console.log("Lecture bloquée par le navigateur :", error); }); } document.addEventListener('click', startMusic); document.addEventListener('touchstart', startMusic);
